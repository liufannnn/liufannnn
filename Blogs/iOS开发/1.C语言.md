1. 位运算与位枚举

   ```objc
   // 位枚举
   // 1 << n == 2 ^ n次方
   int a = 1 << 0; // 1
   int b = 1 << 1; // 2
   int c = 1 << 2; // 4
   int d = 1 << 3; // 8

   // 位运算
   // 枚举中位运算都可以使用或运算
   int value = a | c;

   int compare(int value) {
      // 怎么判断并运算中，有没有包含某个值，直接用&，只要结果不为0就包含
      if (value & a) NSLog(@"包含了a");
      if (value & b) NSLog(@"包含了b");
      if (value & c) NSLog(@"包含了c");
      if (value & d) NSLog(@"包含了d");
   }

   compare(value) 
   // 包含了a
   // 包含了c
   ```
